Behind every prompt is an API call like this one:
 const fs = require("fs"); const ModelSDK = require("modelSDK"); const modelSDK = new ModelSDK({ 
  apiKey: process.env.MODEL_API_KEY,
 }); const htmlContent = fs.readFileSync("index.html", "utf8"); modelSDK.chat.completions.create({ 
  model: "model-name",
 
  messages: [
 
    {
 
      role: "system",
 
      content: "You will be provided with an HTML file with English content,
 
                and your task is to translate only the content into Spanish and return valid HTML.",
 
    },
 
    {
 
      role: "user",
 
      content: htmlContent,
 
    },
 
  ],
 
  temperature: 0.7,
 
  max_tokens: 8192,
 
  top_p: 1,
 }).then(response => { const messageContent = response.choices[0].message.content; 
  fs.writeFileSync("index_es.html", messageContent, "utf8");
 
  console.log("Translation saved as index_es.html");
 }).catch(error => { 
  console.error(error);
 });
